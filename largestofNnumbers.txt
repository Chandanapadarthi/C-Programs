// Print largest of N numbers

#include <stdio.h>

void main(){
	int count;
	printf("Enter how many numbers you want to read: ");
	scnaf("%d", &count);
	int numbers[count];
	for(int i = 0; i < count; i++) {
		scanf("%d", &numbers[i]);
	}
	int largest = 0;
	for(int i = 0; i < count; i++) {
		if(largest < numbers[i]) {
			largest = numbers[i];
		}
	}
	printf("The largest of the given %d numbers is %d", count, largest);
}
